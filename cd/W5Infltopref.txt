int main(int argc, char* argv[])                                                                                                                                        
{

    {                                                                                                                                                                   
        FILE *fp = fopen(argv[1], "r");

        {

    }

void pop()                                                                                                                                                              
{                                                                                                                                                                       
    tos--;                                                                                                                                                              
}                                                                                                                                                                       
char top()                                                                                                                                                              
{                                                                                                                                                                       
    return stack[tos-1];                                                                                                                                                
}                                                                                                                                                                       
void push(char c)                                                                                                                                                       
{                                                                                                                                                                       
    stack[tos]=c;                                                                                                                                                       
    tos++;                                                                                                                                                              
}                                                                                                                                                                       
int priority(char c)                                                                                                                                                    
{                                                                                                                                                                       
    if(c=='+')                                                                                                                                                          
    {                                                                                                                                                                   
        return 1;                                                                                                                                                       
    }                                                                                                                                                                   
    else if(c=='-')                                                                                                                                                     
    {                                                                                                                                                                   
        return 1;                                                                                                                                                       
    }                                                                                                                                                                   
    else if(c==')')                                                                                                                                                     
    {                                                                                                                                                                   
        return 0;                                                                                                                                                       
    }                                                                                                                                                                   
    else if(c=='(')                                                                                                                                                     
    {                                                                                                                                                                   
        return 0;                                                                                                                                                       
    }                                                                                                                                                                   
    else if(c=='*')                                                                                                                                                     
    {                                                                                                                                                                   
        return 2;                                                                                                                                                       
    }                                                                                                                                                                   
    else if(c=='/')                                                                                                                                                     
    {                                                                                                                                                                   
        return 2;                                                                                                                                                       
    }                                                                                                                                                                   
    else if(c=='^')                                                                                                                                                     
    {                                                                                                                                                                   
        return 3;                                                                                                                                                       
    }                                                                                                                                                                   
}
